def series_to_supervised(data, n_in=1, n_out=1, dropnan=True):
    n_vars = 1 if type(data) is list else data.shape[1]
    dff = pd.DataFrame(data)
    cols, names = list(), list()
    
    # input sequence (t-n, ... t-1)
    for i in range(n_in, 0, -1):
        cols.append(dff.shift(i))
        names += [('var%d(t-%d)' % (j+1, i)) for j in range(n_vars)]
    
    # forecast sequence (t, t+1, ... t+n)
    for i in range(0, n_out):
        cols.append(dff.shift(-i))
        if i == 0:
            names += [('var%d(t)' % (j+1)) for j in range(n_vars)]
        else:
            names += [('var%d(t+%d)' % (j+1, i)) for j in range(n_vars)]

    # put it all together
    agg = pd.concat(cols, axis=1)
    agg.columns = names
    # drop rows with NaN values
    if dropnan:
        agg.dropna(inplace=True)
    
    return agg

def get_rnn_dataset(data, past_history=10, forecast_horizon=1):
    n_vars = data.shape[1]
    values = series_to_supervised(data, past_history, forecast_horizon).values
    train_X, train_y = values[:, :-1*n_vars*forecast_horizon], values[:, n_vars*past_history+2::n_vars]
    
    return train_X, train_y

class Scaler:
    def fit(self, data):
        self._min = data.min()
        self._max = data.max()
    
    def normalize(self, data):
        return (data - self._min) / (self._max - self._min)
    
    def denormalize(self, data):
        return data*(self._max-self._min)+self._min

def plot_result(x, y, y_pred=None):
    fig, ax = plt.subplots(1, 1, figsize=(15,5))
    ax.plot(x[0][2::5], 'o-', c='steelblue', ms=3.5, label='History')
    ax.plot([m for m in range(int(x.shape[1]/5), 
                              int(x.shape[1]/5+y.shape[0]))], y, 'x-', c='mediumseagreen', ms=3.5, label='Target')
    
    if y_pred is not None:
        ax.plot([m for m in range(int(x.shape[1]/5), 
                              int(x.shape[1]/5+y.shape[0]))], y_pred, 'o-', c='salmon', ms=3.5, label='Predict')
        
    ax.legend()
